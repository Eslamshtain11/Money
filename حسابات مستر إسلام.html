    function updateDashboardCards() {
        const now = new Date();
        const currentMonth = now.getMonth();
        const currentYear = now.getFullYear();

        const monthlyPayments = allPayments.filter(p => {
            const paymentDate = new Date(p.date);
            return paymentDate.getMonth() === currentMonth && paymentDate.getFullYear() === currentYear;
        });

        const totalThisMonth = monthlyPayments.reduce((sum, p) => sum + p.amount, 0);
        document.getElementById('total-this-month').textContent = `${totalThisMonth.toLocaleString()} جنيه`;

        document.getElementById('total-payments-count').textContent = allPayments.length;

        const monthlyTotals = {};
        allPayments.forEach(p => {
            const monthYear = new Date(p.date).toLocaleDateString('ar-EG', { month: 'long', year: 'numeric' });
            monthlyTotals[monthYear] = (monthlyTotals[monthYear] || 0) + p.amount;
        });

        const totalsArray = Object.entries(monthlyTotals);
        if (totalsArray.length > 0) {
            const highest = totalsArray.reduce((max, month) => month[1] > max[1] ? month : max);
            const lowest = totalsArray.reduce((min, month) => month[1] < min[1] ? month : min);
            document.getElementById('highest-month').textContent = `${highest[0]} (${highest[1].toLocaleString()} ج)`;
            document.getElementById('lowest-month').textContent = `${lowest[0]} (${lowest[1].toLocaleString()} ج)`;
        } else {
             document.getElementById('highest-month').textContent = '-';
             document.getElementById('lowest-month').textContent = '-';
        }
    }

        XLSX.writeFile(workbook, `كشف_شهر_${now.toLocaleDateString('ar-EG', {month: 'long'})}.xlsx`);

    document.getElementById('export-pdf-monthly').addEventListener('click', async () => {
        const now = new Date();
        const currentMonth = now.getMonth();
        const currentYear = now.getFullYear();

        const monthlyPayments = allPayments.filter(p => {
            const paymentDate = new Date(p.date);
            return paymentDate.getMonth() === currentMonth && paymentDate.getFullYear() === currentYear;
        });

        if (monthlyPayments.length === 0) {
            showNotification('لا توجد بيانات لطباعة تقرير هذا الشهر.', 'error');
            return;
        }

        const total = monthlyPayments.reduce((sum, p) => sum + p.amount, 0);

        // Populate the hidden PDF template
        document.getElementById('pdf-month-name').textContent = now.toLocaleDateString('ar-EG', { month: 'long', year: 'numeric' });
        document.getElementById('pdf-month-total').textContent = `${total.toLocaleString()} جنيه`;
        document.getElementById('pdf-report-date').textContent = now.toLocaleDateString('ar-EG');

        const pdfTableBody = document.getElementById('pdf-table-body');
        pdfTableBody.innerHTML = monthlyPayments.map(p => `
            <tr>
                <td class="p-2 border border-gray-400">${p.studentName}</td>
                <td class="p-2 border border-gray-400">${p.groupName}</td>
                <td class="p-2 border border-gray-400">${p.amount.toLocaleString()}</td>
                <td class="p-2 border border-gray-400">${new Date(p.date).toLocaleDateString('ar-EG')}</td>
            </tr>
        `).join('');

        // Generate a temporary chart for the PDF
        const pdfChartDataUrl = monthlyIncomeChart.toBase64Image();
        document.getElementById('pdf-chart-image').src = pdfChartDataUrl;

        const { jsPDF } = window.jspdf;
        const pdf = new jsPDF('p', 'mm', 'a4');
        const reportElement = document.getElementById('pdf-report-template');

        const canvas = await html2canvas(reportElement, {
            scale: 2, // Improve quality
            useCORS: true,
            allowTaint: true
        });

        const imgData = canvas.toDataURL('image/png');
        const imgProps = pdf.getImageProperties(imgData);
        const pdfWidth = pdf.internal.pageSize.getWidth();
        const pdfHeight = (imgProps.height * pdfWidth) / imgProps.width;

        pdf.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);
        pdf.save(`تقرير_شهر_${now.toLocaleDateString('ar-EG', {month: 'long'})}.pdf`);
    });
